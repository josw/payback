CREATE TABLE User (
     userId     TEXT PRIMARY KEY,
     password     TEXT,
     regdate     DATE
);

CREATE TABLE DebtHistory (
     creditor	TEXT,
     debtor 	TEXT,
	 amount	    NUMERIC,
	 reason		TEXT,
     regdate    DATE
);

CREATE TABLE DebtDependency (
     creditor	TEXT PRIMARY KEY,
     debtor 	TEXT,
	 amount	    NUMERIC
);
