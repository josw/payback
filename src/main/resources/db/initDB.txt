CREATE TABLE User (
     userId     TEXT PRIMARY KEY,
     password     TEXT,
     regdate     TEXT
);

CREATE TABLE DebtHistory (
     creditor	TEXT,
     debtor 	TEXT,
	 amount	    NUMERIC,
	 reason		TEXT,
	 complete  NUMERIC DEFAULT 0,
     regdate    TEXT
);

CREATE TABLE DebtDependency (
     creditor	TEXT,
     debtor 	TEXT,
	 amount	    NUMERIC
);
